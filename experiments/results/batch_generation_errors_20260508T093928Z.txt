std::vector<uint8_t> des_encrypt_ofb(const std::vector<uint8_t>& plaintext, const std::vector<uint8_t>& key, std::vector<uint8_t>& iv) {
    std::vector<uint8_t> ciphertext(plaintext.size());
    std::vector<uint8_t> feedback = iv;
    std::vector<uint8_t> block(8);

    for (size_t i = 0; i < plaintext.size(); i++) {
        std::copy(feedback.begin(), feedback.end(), block.begin());
        for (int j = 0; j < 8; ++j) {
            block[j] ^= key[j % DES_KEY_SIZE];
