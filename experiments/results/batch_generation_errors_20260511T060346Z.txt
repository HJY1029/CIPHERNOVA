// 固定参数
static const uint32_t FK[4] = {0xa3b1bac6, 0x56aa3350, 0x677d9197, 0xb27022dc};
static const uint32_t CK[32] = {
    0x00070e15, 0x1c232a31, 0x383f464d, 0x545b6269, 0x70777e85, 0x8c939aa1, 0xa8afb6bd, 0xc4cbd2d9,

};

// 循环左移
uint32_t rotate_left(uint32_t x, int bits) {
    return (x << bits) | (x >> (32 - bits));
}

// T函数
uint32_t T(uint32_t x) {
    uint8_t b0 = Sbox[(x >> 24) & 0xF][(x >> 20) & 0xF];
    uint8_t b1 = Sbox[(x >> 16) & 0xF][(x >> 12) & 0xF];
    uint8_t b2 = Sbox[(x >> 8) & 0xF][(x >> 4) & 0xF];
    uint8_t b3 = Sbox[x & 0xF][x & 0xF];
    uint32_t t = (b0 << 24) | (b1 << 16) | (b2 << 8) | b3;
    return t ^ rotate_left(t, 2) ^ rotate_left(t, 10) ^ rotate_left(t, 18) ^ rotate_left(t, 24);
}

// 密钥扩展
void key_expansion(uint32_t MK[], uint32_t rk[]) {
    uint32_t K[4];
    K[0] = MK[0] ^ FK[0];
    K[1] = MK[1] ^ FK[1];
    K[2] = MK[2] ^ FK[2];
    K[3] = MK[3] ^ FK[3];
    for (int i = 0; i < 32; i++) {
        K[i % 4] = K[i % 4] ^ T(K[(i + 1) % 4] ^ K[(i + 2) % 4] ^ K[(i + 3) % 4] ^ CK[i]);
        rk[i] = K[i % 4];
    }
}

// SM4加密一个分组
void sm4_encrypt_block(uint8_t input[16], uint8_t output[16], uint32_t rk[32]) {
    uint32_t X[4], tmp;
    for (int i = 0; i < 4; i++) {
        X[i] = (input[4 * i] << 24) | (input[4 * i + 1] << 16) | (input[4 * i + 2] << 8) | input[4 * i + 3];
    }
    for (int i = 0; i < 32; i++) {
        tmp = X[(i + 1) % 4] ^ X[(i + 2) % 4] ^ X[(i + 3) % 4] ^ rk[i];
        tmp = T(tmp);
        X[i % 4] ^= tmp;
    }
    for (int i = 0; i < 4; i++) {
        output[4 * i] = (X[3 - i] >> 24) & 0xFF;
        output[4 * i + 1] = (X[3 - i] >> 16) & 0xFF;
        output[4 * i + 2] = (X[3 - i] >> 8) & 0xFF;
        output[4 * i + 3] = X[3 - i] & 0xFF;
    }
}

// OFB模式加密
void sm4_ofb_encrypt(uint8_t *plaintext, int plaintext_len, uint8_t *key, uint8_t *iv, uint8_t *ciphertext) {
    uint32_t rk[32];
    uint8_t feedback[16];
    memcpy(feedback, iv, 16);
    key_expansion((uint32_t *)key, rk);
    int i = 0;
    while (i < plaintext_len) {
        uint8_t keystream[16];
        sm4_encrypt_block(feedback, keystream, rk);
        for (int j = 0; j < 16 && i < plaintext_len; j++, i++) {
            ciphertext[i] = plaintext[i] ^ keystream[j];
        }
        memcpy(feedback, keystream, 16);
    }
}
