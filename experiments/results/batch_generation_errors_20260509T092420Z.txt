unsigned int rotate_left(unsigned int x, int n) {
    return (x << n) | (x >> (32 - n));
}

unsigned int L_prime(unsigned int B) {
    return B ^ rotate_left(B, 13) ^ rotate_left(B, 23);
}

unsigned int tau(unsigned int A) {
    return (SBOX_TABLE[A >> 24] << 24) | (SBOX_TABLE[(A >> 16) & 0xFF] << 16) |
           (SBOX_TABLE[(A >> 8) & 0xFF] << 8) | SBOX_TABLE[A & 0xFF];
}

void key_expansion(unsigned char key[16], unsigned int rk[32]) {
    unsigned int K[36];
    int i;
    K[0] = *(unsigned int *)(key) ^ FK[0];
    K[1] = *(unsigned int *)(key + 4) ^ FK[1];
    K[2] = *(unsigned int *)(key + 8) ^ FK[2];
    K[3] = *(unsigned int *)(key + 12) ^ FK[3];
    for (i = 0; i < 32; i++) {
        K[i + 4] = K[i] ^ L_prime(tau(K[i + 1] ^ K[i + 2] ^ K[i + 3] ^ CK[i]));
        rk[i] = K[i + 4];
    }
}

void sm4_encrypt(unsigned char input[16], unsigned char output[16], unsigned int rk[32]) {
    unsigned int X[36], tmp;
    int i;
    for (i = 0; i < 4; i++) {
        X[i] = *(unsigned int *)(input + i * 4);
    }
    for (i = 0; i < 32; i++) {
        tmp = X[i] ^ L_prime(tau(X[i + 1] ^ X[i + 2] ^ X[i + 3] ^ rk[i]));
        X[i + 4] = tmp;
    }
    for (i = 0; i < 4; i++) {
        *(unsigned int *)(output + i * 4) = X[35 - i];
    }
}

void sm4_ofb(unsigned char *plaintext, unsigned char *ciphertext, int length, unsigned char *key, unsigned char *iv) {
    unsigned int rk[32];
    unsigned char feedback[16];
    int i, j;

    key_expansion(key, rk);
    memcpy(feedback, iv, 16);

    for (i = 0; i < length; i += 16) {
        sm4_encrypt(feedback, feedback, rk);
        for (j = 0; j < 16 && (i + j) < length; j++) {
            ciphertext[i + j] = plaintext[i + j] ^ feedback[j];
        }
    }
}
