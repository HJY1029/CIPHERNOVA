// SM4-CFB模式加密 - CFB-128位模式

    sm4_generate_round_keys(key.data(), rk);
    
    // 初始化反馈寄存器（复制IV）
    uint8_t feedback[16];
    std::copy(iv.begin(), iv.end(), feedback);
    
    // 分配输出缓冲区
    std::vector<uint8_t> ciphertext(plaintext.size());
    
    // CFB-128模式加密 - 每次处理一个完整块

        // 加密反馈寄存器
        uint8_t encrypted_feedback[16];
        sm4_encrypt_block(rk, feedback, encrypted_feedback);
        
        // 处理当前块
        size_t block_size = std::min((size_t)16, plaintext.size() - i);
        for (size_t j = 0; j < block_size; ++j) {
            ciphertext[i + j] = plaintext[i + j] ^ encrypted_feedback[j];
        }
        
        // 更新反馈寄存器（左移一个块大小，将新的密文移入）
        memmove(feedback, feedback + block_size, 16 - block_size);
        std::copy(&ciphertext[i], &ciphertext[i + block_size], feedback + (16 - block_size));
    }
    
    return ciphertext;
}
