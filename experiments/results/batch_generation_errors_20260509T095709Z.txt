// 循环左移
unsigned int rotate_left(unsigned int x, int bits) {
    return (x << bits) | (x >> (32 - bits));
}

// T-变换
unsigned int T(unsigned int x) {
    return (SBOX[(x >> 24) & 0xFF] << 24) | (SBOX[(x >> 16) & 0xFF] << 16) | (SBOX[(x >> 8) & 0xFF] << 8) | SBOX[x & 0xFF];
}

// L-变换
unsigned int L(unsigned int B) {
    return B ^ rotate_left(B, 2) ^ rotate_left(B, 10) ^ rotate_left(B, 18) ^ rotate_left(B, 24);
}

// 密钥扩展
void key_expansion(unsigned int MK[], unsigned int rk[]) {
    unsigned int K[4];
    K[0] = MK[0] ^ FK[0];
    K[1] = MK[1] ^ FK[1];
    K[2] = MK[2] ^ FK[2];
    K[3] = MK[3] ^ FK[3];
    for (int i = 0; i < 32; i++) {
        K[i % 4] = K[i % 4] ^ L(T(K[(i + 1) % 4] ^ K[(i + 2) % 4] ^ K[(i + 3) % 4] ^ CK[i]));
        rk[i] = K[i % 4];
    }
}

// SM4加密一个分组
void sm4_encrypt_block(unsigned char input[16], unsigned char output[16], unsigned int rk[32]) {
    unsigned int X[4], tmp;
    for (int i = 0; i < 4; i++) {
        X[i] = (input[4 * i] << 24) | (input[4 * i + 1] << 16) | (input[4 * i + 2] << 8) | input[4 * i + 3];
    }
    for (int i = 0; i < 32; i++) {
        tmp = X[(i + 1) % 4] ^ X[(i + 2) % 4] ^ X[(i + 3) % 4] ^ rk[i];
        X[i % 4] ^= L(T(tmp));
    }
    for (int i = 0; i < 4; i++) {
        output[4 * i] = (X[3 - i] >> 24) & 0xFF;
        output[4 * i + 1] = (X[3 - i] >> 16) & 0xFF;
        output[4 * i + 2] = (X[3 - i] >> 8) & 0xFF;
        output[4 * i + 3] = X[3 - i] & 0xFF;
    }
}

// 从十六进制字符串转换为字节数组
int hex_to_bytes(const char *hex, unsigned char *bytes, int max_bytes) {
    int num_bytes = 0;
    while (*hex && *(hex + 1) && num_bytes < max_bytes) {
        sscanf(hex, "%2hhx", &bytes[num_bytes]);
        num_bytes++;
        hex += 2;
    }
    return num_bytes;
}

// 去除字符串中的所有空白字符
void remove_whitespace(char *str) {
    char *p1 = str, *p2 = str;
    while (*p2 != '\0') {
        if (!isspace((unsigned char)*p2)) {
            *p1++ = *p2;
        }
        p2++;
    }
    *p1 = '\0';
}
