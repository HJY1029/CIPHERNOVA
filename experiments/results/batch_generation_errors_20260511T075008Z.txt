        for (j = 0; j < 4; j++) {
            state[j][i] = in[i * 4 + j];
        }
    }

    // Add the First round key to the state before starting the rounds.
    AddRoundKey(state, w[0]);

    // There will be Nr rounds.
    // The first Nr-1 rounds are identical.
    // These Nr-1 rounds are executed in the loop below.
    for (round = 1; round < Nr; round++) {
        SubBytes(state);
        ShiftRows(state);
        MixColumns(state);
        AddRoundKey(state, w[round]);

    // The MixColumns function is not here in the last round.
    SubBytes(state);
    ShiftRows(state);
    AddRoundKey(state, w[Nr]);

    // The encryption process is over.
    // Copy the state array to output array.
    for (i = 0; i < 4; i++) {
        for (j = 0; j < 4; j++) {
            out[i * 4 + j] = state[j][i];
        }
    }
}

    memcpy(feedback, iv, AES_BLOCK_SIZE);

    // Encrypt plaintext block by block
    for (i = 0; i < length; i += AES_BLOCK_SIZE) {
        // Encrypt the feedback register
        Cipher(feedback, temp, w);

        // XOR plaintext with the encrypted feedback to produce ciphertext
        for (j = 0; j < AES_BLOCK_SIZE && (i + j) < length; j++) {
            ciphertext[i + j] = plaintext[i + j] ^ temp[j];
        }

        // Update feedback register
        memcpy(feedback, temp, AES_BLOCK_SIZE);
    }
}
