void des_cfb_encrypt(const unsigned char *key, const unsigned char *iv, const unsigned char *plaintext, unsigned char *ciphertext, size_t length) {
    DES_key_schedule schedule;
    DES_set_odd_parity(key);
    DES_set_key_unchecked((DES_cblock *)key, &schedule);

    unsigned char current_iv[IV_SIZE];
    memcpy(current_iv, iv, IV_SIZE);

    for (size_t i = 0; i < length; i++) {
        DES_ecb_encrypt((DES_cblock *)current_iv, (DES_cblock *)&ciphertext[i], &schedule, DES_ENCRYPT);
        ciphertext[i] ^= plaintext[i];
        memmove(current_iv, current_iv + 1, IV_SIZE - 1);
        current_iv[IV_SIZE - 1] = ciphertext[i];
    }
}

int main() {
    char *plaintext_hex = getenv("TEST_PLAINTEXT");
    char *key_hex = getenv("TEST_KEY");
    char *iv_hex = getenv("TEST_IV");

    if (!plaintext_hex || !key_hex) {
        fprintf(stderr, "Missing environment variables\n");
        return 1;
    }

    remove_whitespace(plaintext_hex);
    remove_whitespace(key_hex);

    size_t plaintext_len = strlen(plaintext_hex) / 2;
    size_t key_len = strlen(key_hex) / 2;

    if (key_len != KEY_SIZE || plaintext_len % IV_SIZE != 0) {
        fprintf(stderr, "Invalid input length\n");
        return 1;
    }

    unsigned char *plaintext = malloc(plaintext_len);
    unsigned char *key = malloc(key_len);
    unsigned char *iv = malloc(IV_SIZE);

    if (hex_to_bytes(plaintext_hex, plaintext, plaintext_len) != 0 ||
        hex_to_bytes(key_hex, key, key_len) != 0) {
        fprintf(stderr, "Invalid hexadecimal input\n");
        free(plaintext);
        free(key);
        free(iv);
        return 1;
    }

    if (iv_hex) {
        remove_whitespace(iv_hex);
        size_t iv_len = strlen(iv_hex) / 2;

        if (iv_len != IV_SIZE) {
            fprintf(stderr, "Invalid IV length\n");
            free(plaintext);
            free(key);
            free(iv);
            return 1;
        }

        hex_to_bytes(iv_hex, iv, IV_SIZE);
    } else {
        memset(iv, 0, IV_SIZE);
    }

    unsigned char *ciphertext = malloc(plaintext_len);

    des_cfb_encrypt(key, iv, plaintext, ciphertext, plaintext_len);

    printf("密文: %.*s\n", (int)plaintext_len, (char *)ciphertext);

    free(plaintext);
    free(key);
    free(iv);
    free(ciphertext);

    return 0;
}
