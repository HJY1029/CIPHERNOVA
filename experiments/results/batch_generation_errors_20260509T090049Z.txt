unsigned int T(unsigned int input) {
    unsigned char a = SBOX[(input >> 24) & 0xFF];
    unsigned char b = SBOX[(input >> 16) & 0xFF];
    unsigned char c = SBOX[(input >> 8) & 0xFF];
    unsigned char d = SBOX[input & 0xFF];
    unsigned int result = (a << 24) | (b << 16) | (c << 8) | d;
    return result ^ (result << 2) ^ (result << 10) ^ (result << 18) ^ (result << 24);
}

void key_expansion(const std::vector<unsigned char>& key, std::vector<unsigned int>& round_keys) {
    unsigned int K[4];
    for (int i = 0; i < 4; i++) {
        K[i] = (key[4 * i] << 24) | (key[4 * i + 1] << 16) | (key[4 * i + 2] << 8) | key[4 * i + 3];
        K[i] ^= FK[i];
    }
    for (int i = 0; i < 32; i++) {
        unsigned int temp = K[i % 4] ^ T(K[(i + 1) % 4] ^ K[(i + 2) % 4] ^ K[(i + 3) % 4] ^ CK[i]);
        round_keys.push_back(temp);
        K[i % 4] = temp;
    }
}

void sm4_round(const std::vector<unsigned char>& input, std::vector<unsigned char>& output, const std::vector<unsigned int>& round_keys) {
    unsigned int X[4];
    for (int i = 0; i < 4; i++) {
        X[i] = (input[4 * i] << 24) | (input[4 * i + 1] << 16) | (input[4 * i + 2] << 8) | input[4 * i + 3];
    }
    for (int i = 0; i < 32; i++) {
        unsigned int temp = X[i % 4] ^ T(X[(i + 1) % 4] ^ X[(i + 2) % 4] ^ X[(i + 3) % 4] ^ round_keys[i]);
        X[i % 4] = temp;
    }
    for (int i = 0; i < 4; i++) {
        output[4 * i] = (X[i] >> 24) & 0xFF;
        output[4 * i + 1] = (X[i] >> 16) & 0xFF;
        output[4 * i + 2] = (X[i] >> 8) & 0xFF;
        output[4 * i + 3] = X[i] & 0xFF;
    }
}

void encrypt_sm4_ofb(const std::vector<unsigned char> &plaintext, const std::vector<unsigned char> &key, const std::vector<unsigned char> &iv, std::vector<unsigned char> &ciphertext) {
    std::vector<unsigned int> round_keys;
    key_expansion(key, round_keys);
    
    std::vector<unsigned char> feedback = iv;
    std::vector<unsigned char> block(16, 0);
    ciphertext.resize(plaintext.size());

    for (size_t i = 0; i < plaintext.size(); i++) {
        sm4_round(feedback, block, round_keys);
        ciphertext[i] = plaintext[i] ^ block[i % 16];
        feedback = block;
    }
}
