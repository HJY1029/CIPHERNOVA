    std::vector<unsigned char> bytes;

void removeWhitespace(std::string &input) {
    input.erase(std::remove_if(input.begin(), input.end(), [](unsigned char ch) {
        return std::isspace(ch) != 0;
    }), input.end());
}

bool readEnvOrStdin(const char *envName, std::string &output) {
    const char *envValue = std::getenv(envName);
    if (envValue != nullptr) {
        output.assign(envValue);
        removeWhitespace(output);
        return !output.empty();
    }
    static bool stdinScanned = false;
    static std::vector<std::string> stdinTokens;
    if (!stdinScanned) {
        stdinScanned = true;
        std::string line;
        while (std::getline(std::cin, line)) {
            removeWhitespace(line);
            if (!line.empty()) {
                stdinTokens.push_back(line);
            }
        }
    }
    if (!stdinTokens.empty()) {
        output.assign(stdinTokens.front());
        stdinTokens.erase(stdinTokens.begin());
        return true;
    }
    return false;
}

bool hexToBytes(const std::string &hex, std::vector<unsigned char> &bytes) {
    if (hex.size() % 2 != 0) {
        return false;
    }
    bytes.clear();
    bytes.reserve(hex.size() / 2);
    for (size_t i = 0; i < hex.size(); i += 2) {
        int high = hexCharToValue(hex[i]);
        int low = hexCharToValue(hex[i + 1]);
        if (high < 0 || low < 0) {
            return false;
        }
        bytes.push_back(static_cast<unsigned char>((high << 4) | static_cast<unsigned char>(low)));
    }
    return true;
}
