#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#define DES_KEY_SIZE 8
#define IV_SIZE 8

void remove_whitespace(char *str) {
    char *src = str, *dst = str;
    while (*src) {
        if (!isspace((unsigned char)*src)) {
            *dst++ = *src;
        }
        src++;
    }
    *dst = '\0';
}

int hex_to_byte(const char *hex, unsigned char *byte) {
    if (strlen(hex) != 2)
        return -1;

    int high = toupper(hex[0]) >= 'A' ? toupper(hex[0]) - 'A' + 10 : toupper(hex[0]) - '0';
    int low = toupper(hex[1]) >= 'A' ? toupper(hex[1]) - 'A' + 10 : toupper(hex[1]) - '0';

    *byte = (high << 4) | low;
    return 0;
}

int hex_to_bytes(const char *hex, unsigned char *bytes, size_t len) {
    for (size_t i = 0; i < len; i++) {
        if (hex_to_byte(hex + i * 2, &bytes[i]) != 0)
            return -1;
    }
    return 0;
}

void des_cfb64_encrypt(const unsigned char *key, const unsigned char *iv, size_t key_len, size_t iv_len, size_t plaintext_len, const unsigned char *plaintext, unsigned char *ciphertext) {
    unsigned char feedback[IV_SIZE];
    memcpy(feedback, iv, IV_SIZE);

    for (size_t i = 0; i < plaintext_len; i++) {
        DES_cfb64_encrypt(key, feedback, key_len, iv_len, &feedback, &ciphertext[i], DES_ENCRYPT);
        ciphertext[i] ^= feedback[0];
    }
}
