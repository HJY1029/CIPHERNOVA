int main() {
    const char *key_str = getenv("TEST_KEY");
    const char *iv_str = getenv("TEST_IV");
    if (!key_str || !iv_str) {
        fprintf(stderr, "Missing TEST_KEY or TEST_IV environment variables\n");
        return 1;
    }

    unsigned char key[16];
    unsigned char iv[8];
    memcpy(key, key_str, 16);
    memcpy(iv, iv_str, 8);

    EVP_CIPHER_CTX *ctx = EVP_CIPHER_CTX_new();
    if (!ctx) {
        fprintf(stderr, "Failed to create cipher context\n");
        return 1;
    }

    if (1 != EVP_EncryptInit_ex(ctx, EVP_sm4_ofb(), NULL, key, iv)) {
        fprintf(stderr, "Failed to initialize encryption\n");
        EVP_CIPHER_CTX_free(ctx);
        return 1;
    }

    const char *plaintext = "your_plaintext_here";
    int plaintext_len = strlen(plaintext);
    unsigned char ciphertext[plaintext_len + EVP_CIPHER_block_size(EVP_sm4_ofb())];
    int ciphertext_len;

    if (1 != EVP_EncryptUpdate(ctx, ciphertext, &ciphertext_len, (unsigned char *)plaintext, plaintext_len)) {
        fprintf(stderr, "Failed to encrypt data\n");
        EVP_CIPHER_CTX_free(ctx);
        return 1;
    }

    int final_len;
    if (1 != EVP_EncryptFinal_ex(ctx, ciphertext + ciphertext_len, &final_len)) {
        fprintf(stderr, "Failed to finalize encryption\n");
        EVP_CIPHER_CTX_free(ctx);
        return 1;
    }

    ciphertext_len += final_len;

    printf("密文: %.*s\n", ciphertext_len, (char *)ciphertext);

    EVP_CIPHER_CTX_free(ctx);
    return 0;
}
