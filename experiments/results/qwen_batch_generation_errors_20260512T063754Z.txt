#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#define DES_BLOCK_SIZE 8

void des_encrypt(const unsigned char *key, const unsigned char *plaintext, unsigned char *ciphertext) {
    DES_key_schedule key_schedule;
    DES_set_key((DES_cblock *)key, &key_schedule);

    for (int i = 0; i < strlen(plaintext); i += DES_BLOCK_SIZE) {
        DES_ecb_encrypt((DES_cblock *)(plaintext + i), (DES_cblock *)(ciphertext + i), &key_schedule, DES_ENCRYPT);
    }
}

void remove_whitespace(char *str) {
    char *src = str;
    char *dst = str;

    while (*src != '\0') {
        if (!isspace((unsigned char)*src)) {
            *dst++ = *src;
        }
        src++;
    }

    *dst = '\0';
}

int main() {
    char key[17] = {0};
    char iv[17] = {0};
    char plaintext[256] = {0};
    unsigned char ciphertext[256] = {0};

    // Read key and IV from environment variables
    if (getenv("TEST_KEY") != NULL) {
        strcpy(key, getenv("TEST_KEY"));
    }
    if (getenv("TEST_IV") != NULL) {
        strcpy(iv, getenv("TEST_IV"));
    }

    // Remove whitespace from key and IV
    remove_whitespace(key);
    remove_whitespace(iv);

    // Read plaintext from environment variable or stdin
    if (getenv("TEST_PLAINTEXT") != NULL) {
        strcpy(plaintext, getenv("TEST_PLAINTEXT"));
    } else {
        printf("Enter plaintext in hex format: ");
        fgets(plaintext, sizeof(plaintext), stdin);
    }

    remove_whitespace(plaintext);

    // Convert key and IV from hex to binary
    for (int i = 0; i < strlen(key); i += 2) {
        sscanf(&key[i], "%2hhx", &key[i / 2]);
    }
    for (int i = 0; i < strlen(iv); i += 2) {
        sscanf(&iv[i], "%2hhx", &iv[i / 2]);
    }

    // Convert plaintext from hex to binary
    for (int i = 0; i < strlen(plaintext); i += 2) {
        sscanf(&plaintext[i], "%2hhx", &plaintext[i / 2]);
    }

    // Pad plaintext to be a multiple of DES block size
    int padding_length = DES_BLOCK_SIZE - (strlen(plaintext) % DES_BLOCK_SIZE);
    if (padding_length == 0) {
        padding_length = DES_BLOCK_SIZE;
    }
    for (int i = strlen(plaintext); i < strlen(plaintext) + padding_length; i++) {
        plaintext[i] = padding_length;
    }

    // Encrypt plaintext
    des_encrypt(key, plaintext, ciphertext);

    // Convert ciphertext to hex and print
    printf("密文: ");
    for (int i = 0; i < strlen((char *)ciphertext); i++) {
        printf("%02x", ciphertext[i]);
    }
    printf("\n");

    return 0;
}
