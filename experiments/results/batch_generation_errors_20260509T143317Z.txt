    return (x << n) | (x >> (32 - n));
}

// T置换
unsigned int T(unsigned int x) {
    return (Sbox[(x >> 24) & 0xFF] << 24) | (Sbox[(x >> 16) & 0xFF] << 16) | (Sbox[(x >> 8) & 0xFF] << 8) | Sbox[x & 0xFF];
}

// 密钥扩展算法
void key_expansion(const unsigned char *key, unsigned int *rk) {
    unsigned int K[36];
    int i;

    // 将密钥转换为4个unsigned int
    for (i = 0; i < 4; i++) {
        K[i] = (key[4 * i] << 24) | (key[4 * i + 1] << 16) | (key[4 * i + 2] << 8) | key[4 * i + 3];
        K[i] ^= FK[i];
    }

    // 计算轮密钥
    for (i = 0; i < 32; i++) {
        K[i + 4] = K[i] ^ T(rotate_left(K[i + 1] ^ K[i + 2] ^ K[i + 3] ^ CK[i], 13));
        rk[i] = K[i + 4];
    }
}

// SM4单轮加密
unsigned int sm4_round(unsigned int X[4], unsigned int rk) {
    return X[0] ^ T(X[1] ^ X[2] ^ X[3] ^ rk);
}

// SM4加密
void sm4_encrypt(unsigned char *input, unsigned char *output, unsigned int *rk) {
    unsigned int X[36];
    int i;

    // 将输入转换为4个unsigned int
    for (i = 0; i < 4; i++) {
        X[i] = (input[4 * i] << 24) | (input[4 * i + 1] << 16) | (input[4 * i + 2] << 8) | input[4 * i + 3];
    }

    // 32轮加密
    for (i = 0; i < 32; i++) {
        X[i + 4] = sm4_round(X + i, rk[i]);
    }

    // 输出结果
    for (i = 0; i < 4; i++) {
        output[4 * i] = (X[35 - i] >> 24) & 0xFF;
        output[4 * i + 1] = (X[35 - i] >> 16) & 0xFF;
        output[4 * i + 2] = (X[35 - i] >> 8) & 0xFF;
        output[4 * i + 3] = X[35 - i] & 0xFF;
    }
}

// 从十六进制字符串转换为字节数据
static int hex_to_bytes(const char *hex, unsigned char *bytes, int max_bytes) {
    int num_bytes = 0;
    while (*hex && *(hex + 1) && num_bytes < max_bytes) {
        sscanf(hex, "%2hhx", &bytes[num_bytes]);
        hex += 2;
        num_bytes++;
    }
    return num_bytes;
}

// 去除字符串中的所有空白字符
static void remove_whitespace(char *str) {
    char *p1 = str, *p2 = str;
    while (*p1 != 0) {
        if (!isspace((unsigned char)*p1)) {
            *p2++ = *p1;
        }
        p1++;
    }
    *p2 = 0;
}

int main() {
    // 从环境变量获取输入数据
    char *env_plaintext = getenv("TEST_PLAINTEXT");
    char *env_key = getenv("TEST_KEY");
    char *env_iv = getenv("TEST_IV");

    // 处理环境变量中的空白字符
    if (env_plaintext) { remove_whitespace(env_plaintext); }
    if (env_key) { remove_whitespace(env_key); }
    if (env_iv) { remove_whitespace(env_iv); }

    // 设置默认值
    if (!env_plaintext) { env_plaintext = ""; }
    if (!env_key) { env_key = ""; }
    if (!env_iv) { env_iv = "00000000000000000000000000000000"; }

    unsigned char key[16];
    unsigned char iv[16];
    unsigned char plaintext[1024];
    unsigned char ciphertext[1040]; // 留出额外空间以防止溢出

    int key_len = hex_to_bytes(env_key, key, sizeof(key));
    int iv_len = hex_to_bytes(env_iv, iv, sizeof(iv));
    int plaintext_len = hex_to_bytes(env_plaintext, plaintext, sizeof(plaintext));

    if (key_len != 16 || iv_len != 16) {
        fprintf(stderr, "密钥或IV长度错误。\n");
        return 1;
    }

    unsigned int rk[32];
    key_expansion(key, rk);

    unsigned char feedback[16];
    memcpy(feedback, iv, 16);

    // OFB模式加密
    int i, j;
    for (i = 0; i < plaintext_len; i++) {
        unsigned char keystream[16];
        sm4_encrypt(feedback, keystream, rk);
        ciphertext[i] = plaintext[i] ^ keystream[0];
        memmove(feedback, feedback + 1, 15);
        feedback[15] = keystream[0];
    }

    // 输出密文
    printf("密文: ");
    for (i = 0; i < plaintext_len; i++) {
        printf("%02x", ciphertext[i]);
    }
    printf("\n");

    return 0;
}
