std::string bytes_to_hex(const std::vector<uint8_t> &bytes) {
    std::stringstream hex;
    for (size_t i = 0; i < bytes.size(); ++i) {
        hex << std::hex << std::setfill('0') << std::setw(2) << static_cast<int>(bytes[i]);
    }
    return hex.str();
}

void remove_whitespace(std::string &str) {
    str.erase(std::remove_if(str.begin(), str.end(), ::isspace), str.end());
}

std::vector<uint8_t> sm4_ofb_encrypt(const std::vector<uint8_t> &plaintext, const std::vector<uint8_t> &key, const std::vector<uint8_t> &iv) {
    std::vector<uint32_t> round_keys;
    key_expansion(key, round_keys);
    std::vector<uint8_t> output(plaintext.size());
    std::vector<uint8_t> feedback(iv.begin(), iv.end());

    for (size_t i = 0; i < plaintext.size(); ++i) {
        if (i % 16 == 0) {
            sm4_block_encrypt(feedback, round_keys);
        }
        output[i] = plaintext[i] ^ feedback[i % 16];
    }
    return output;
}

int main() {
    const char *plaintext_hex = std::getenv("TEST_PLAINTEXT");
    const char *key_hex = std::getenv("TEST_KEY");
    const char *iv_hex = std::getenv("TEST_IV");

    if (!plaintext_hex || !key_hex || !iv_hex) {
        std::cerr << "Environment variables TEST_PLAINTEXT, TEST_KEY, or TEST_IV are not set." << std::endl;
        return 1;
    }

    std::string plaintext_str(plaintext_hex);
    std::string key_str(key_hex);
    std::string iv_str(iv_hex);

    remove_whitespace(plaintext_str);
    remove_whitespace(key_str);
    remove_whitespace(iv_str);

    std::vector<uint8_t> plaintext = hex_to_bytes(plaintext_str);
    std::vector<uint8_t> key = hex_to_bytes(key_str);
    std::vector<uint8_t> iv = hex_to_bytes(iv_str);

    if (key.size() != 16 || iv.size() != 16) {
        std::cerr << "Key and IV must be exactly 16 bytes (32 hex characters) long." << std::endl;
        return 1;
    }

    std::vector<uint8_t> ciphertext = sm4_ofb_encrypt(plaintext, key, iv);
    std::cout << "密文: " << bytes_to_hex(ciphertext) << std::endl;

    return 0;
}
