void remove_whitespace(char *str) {
    char *src = str, *dst = str;
    while (*src) {
        if (!isspace((unsigned char)*src)) {
            *dst++ = *src;
        }
        src++;
    }
    *dst = '\0';
}

int hex_to_byte(const char *hex, unsigned char *byte) {
    if (strlen(hex) != 2)
        return -1;
    *byte = (hex[0] >= 'a' ? hex[0] - 'a' + 10 : hex[0] - '0') * 16 +
            (hex[1] >= 'a' ? hex[1] - 'a' + 10 : hex[1] - '0');
    return 0;
}

void byte_to_hex(unsigned char byte, char *hex) {
    hex[0] = byte / 16 < 10 ? byte / 16 + '0' : byte / 16 - 10 + 'a';
    hex[1] = byte % 16 < 10 ? byte % 16 + '0' : byte % 16 - 10 + 'a';
}

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#define DES_KEY_SIZE 8
#define IV_SIZE 8

void remove_spaces(char *str) {
    int i, j;
    for (i = 0, j = 0; str[i]; i++) {
        if (!isspace(str[i])) {
            str[j++] = str[i];
        }
    }
    str[j] = '\0';
}

int hex_to_byte(const char *hex, unsigned char *byte) {
    if (strlen(hex) != 2) return -1;
    *byte = (unsigned char)((hex[0] >= 'A' && hex[0] <= 'F') ? ((hex[0] - 'A') + 10) : (hex[0] - '0')) << 4 |
              ((hex[1] >= 'A' && hex[1] <= 'F') ? ((hex[1] - 'A') + 10) : (hex[1] - '0'));
    return 0;
}

int bytes_to_hex(const unsigned char *bytes, size_t len, char *hex) {
    for (size_t i = 0; i < len; ++i) {
        sprintf(hex + i * 2, "%02x", bytes[i]);
    }
    hex[len * 2] = '\0';
    return 0;
}

void des_cfb_encrypt(const unsigned char *key, const unsigned char *iv, const unsigned char *plaintext, size_t plaintext_len, unsigned char *ciphertext) {
    unsigned char current_iv[IV_SIZE];
    memcpy(current_iv, iv, IV_SIZE);

    for (size_t i = 0; i < plaintext_len; ++i) {
        unsigned char block[IV_SIZE];
        DES_cfb64_encrypt(current_iv, block, IV_SIZE, key, &key, DES_ENCRYPT);
        ciphertext[i] = plaintext[i] ^ block[i % IV_SIZE];
        memmove(current_iv, current_iv + 1, IV_SIZE - 1);
        current_iv[IV_SIZE - 1] = ciphertext[i];
    }
}

int main() {
    char *plaintext_hex = getenv("TEST_PLAINTEXT");
    char *key_hex = getenv("TEST_KEY");
    char *iv_hex = getenv("TEST_IV");

    if (!plaintext_hex || !key_hex) {
        fprintf(stderr, "Error: Missing environment variables.\n");
        return 1;
    }

    remove_spaces(plaintext_hex);
    remove_spaces(key_hex);

    unsigned char plaintext[DES_KEY_SIZE];
