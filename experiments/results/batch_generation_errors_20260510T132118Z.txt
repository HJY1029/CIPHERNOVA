    for (int i = 0; i < 32; i++) {
        uint32_t temp = X[i + 1] ^ X[i + 2] ^ X[i + 3] ^ rk[i];
        temp = tau(temp);
        X[i + 4] = X[i] ^ L(temp);
    }
    for (int i = 0; i < 4; i++) {
        uint32_t val = X[35 - i];
        output[4 * i] = (val >> 24) & 0xFF;
        output[4 * i + 1] = (val >> 16) & 0xFF;
        output[4 * i + 2] = (val >> 8) & 0xFF;
        output[4 * i + 3] = val & 0xFF;
    }
}

void sm4_ofb_encrypt(uint8_t *plaintext, int plaintext_len, uint8_t *key, uint8_t *iv, uint8_t *ciphertext) {
    uint32_t rk[32];
    uint8_t feedback[16];
    memcpy(feedback, iv, 16);

    key_expansion((uint32_t *)key, rk);

    int num_blocks = (plaintext_len + 15) / 16;
    for (int i = 0; i < num_blocks; i++) {
        uint8_t output_block[16];
        sm4_encrypt_block(rk, feedback, output_block);
        int block_size = (i == num_blocks - 1) ? plaintext_len % 16 : 16;
        if (block_size == 0) block_size = 16;
        for (int j = 0; j < block_size; j++) {
            ciphertext[i * 16 + j] = plaintext[i * 16 + j] ^ output_block[j];
        }
        memcpy(feedback, output_block, 16);
    }
}
