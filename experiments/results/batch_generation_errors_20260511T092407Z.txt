            ciphertext[i * SM4_BLOCK_SIZE + j] = plaintext[i * SM4_BLOCK_SIZE + j] ^ keystream[j];
        }
        memcpy(feedback, keystream, SM4_BLOCK_SIZE);
    }
}

// 十六进制字符串转换为字节流
int hex_to_bytes(const char *hex, unsigned char *bytes, int max_len) {
    int num_bytes = 0;
    while (*hex && *(hex + 1) && num_bytes < max_len) {
        sscanf(hex, "%2hhx", &bytes[num_bytes]);
        hex += 2;
        num_bytes++;
    }
    return num_bytes;
}

int main() {
    char *env_plaintext = getenv("TEST_PLAINTEXT");
    char *env_key = getenv("TEST_KEY");
    char *env_iv = getenv("TEST_IV");

    if (!env_plaintext || !env_key || !env_iv) {
        fprintf(stderr, "环境变量中缺少必要的数据。\n");
        return 1;
    }

    unsigned char plaintext[1024];
    unsigned char key[SM4_KEY_SIZE];
    unsigned char iv[SM4_BLOCK_SIZE];
    int plaintext_len = hex_to_bytes(env_plaintext, plaintext, sizeof(plaintext));
    int key_len = hex_to_bytes(env_key, key, sizeof(key));
    int iv_len = hex_to_bytes(env_iv, iv, sizeof(iv));

    if (key_len != SM4_KEY_SIZE || iv_len != SM4_BLOCK_SIZE) {
        fprintf(stderr, "密钥或IV长度不正确。\n");
        return 1;
    }

    unsigned char ciphertext[1024];
    sm4_ofb_encrypt(plaintext, ciphertext, plaintext_len, key, iv);

    printf("密文: ");
    for (int i = 0; i < plaintext_len; i++) {
        printf("%02x", ciphertext[i]);
    }
    printf("\n");

    return 0;
}
