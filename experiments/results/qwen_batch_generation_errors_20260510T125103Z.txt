    *dst = '\0';
}

void des_cfb_encrypt(const unsigned char *key, const unsigned char *iv, const unsigned char *plaintext, size_t plaintext_len, unsigned char *ciphertext) {
    DES_key_schedule key_schedule;
    DES_set_odd_parity((DES_cblock *)key);
    DES_set_key_checked((DES_cblock *)key, &key_schedule);

    for (size_t i = 0; i < plaintext_len; i++) {
        DES_ede3_cfb64_encrypt(plaintext + i, ciphertext + i, 1, &key_schedule, (DES_cblock *)iv, &i, DES_ENCRYPT);
    }
}

int main() {
    char key_hex[KEY_SIZE * 2 + 1];
    char iv_hex[IV_SIZE * 2 + 1];
    char plaintext_hex[1024];

    // Read from environment variables
    if (getenv("TEST_KEY")) {
        strncpy(key_hex, getenv("TEST_KEY"), sizeof(key_hex));
    } else {
        printf("Key not provided. Exiting.\n");
        return 1;
    }

    if (getenv("TEST_IV")) {
        strncpy(iv_hex, getenv("TEST_IV"), sizeof(iv_hex));
    } else {
        memset(iv_hex, '0', IV_SIZE * 2);
    }

    if (fgets(plaintext_hex, sizeof(plaintext_hex), stdin) == NULL) {
        printf("No input provided. Exiting.\n");
        return 1;
    }

    // Remove whitespace from the key and IV
    remove_whitespace(key_hex);
    remove_whitespace(iv_hex);

    // Convert hex strings to byte arrays
    unsigned char key[KEY_SIZE];
    unsigned char iv[IV_SIZE];
    unsigned char plaintext[(strlen(plaintext_hex) + 1) / 2];
    unsigned char ciphertext[strlen(plaintext_hex) / 2];

    for (size_t i = 0; i < KEY_SIZE * 2; i += 2) {
        sscanf(key_hex + i, "%2hhx", &key[i / 2]);
    }

    for (size_t i = 0; i < IV_SIZE * 2; i += 2) {
        sscanf(iv_hex + i, "%2hhx", &iv[i / 2]);
    }

    size_t plaintext_len = strlen(plaintext_hex);
    for (size_t i = 0; i < plaintext_len; i += 2) {
        sscanf(plaintext_hex + i, "%2hhx", &plaintext[i / 2]);
    }

    // Encrypt the plaintext
    des_cfb_encrypt(key, iv, plaintext, strlen((char *)plaintext), ciphertext);

    // Output the ciphertext in hex format
    printf("密文: ");
    for (size_t i = 0; i < strlen((char *)ciphertext); i++) {
        printf("%02x", ciphertext[i]);
    }
    printf("\n");

    return 0;
}
