// Rcon表
static const uint32_t rcon[10] = {0x01000000, 0x02000000, 0x04000000, 0x08000000, 0x10000000,
                                  0x20000000, 0x40000000, 0x80000000, 0x1b000000, 0x36000000};

// 字循环左移
uint32_t rot_word(uint32_t word) {
    return ((word << 8) | (word >> 24)) & 0xFFFFFFFF;
}

// 字字节替换
uint32_t sub_word(uint32_t word) {
    return ((uint32_t)sbox[(word >> 24) & 0xFF] << 24) |
           ((uint32_t)sbox[(word >> 16) & 0xFF] << 16) |
           ((uint32_t)sbox[(word >> 8) & 0xFF] << 8) |
           ((uint32_t)sbox[word & 0xFF]);
}

// AES密钥扩展
void key_expansion(const uint8_t *key, int key_len, uint32_t *w) {
    int Nk, Nr;
    if (key_len == 16) { Nk = 4; Nr = AES_128_ROUNDS; }
    else if (key_len == 24) { Nk = 6; Nr = AES_192_ROUNDS; }
    else if (key_len == 32) { Nk = 8; Nr = AES_256_ROUNDS; }
    else { return; }

    // 复制原始密钥到w[0..Nk-1]
    for (int i = 0; i < Nk; ++i) {
        w[i] = ((uint32_t)key[4*i] << 24) | ((uint32_t)key[4*i+1] << 16) |
               ((uint32_t)key[4*i+2] << 8) | (uint32_t)key[4*i+3];
    }

    // 生成剩余的轮密钥
    for (int i = Nk; i < (Nr + 1) * 4; ++i) {
        uint32_t temp = w[i - 1];
        if (i % Nk == 0) {
            temp = sub_word(rot_word(temp)) ^ rcon[i / Nk - 1];
        } else if (Nk > 6 && i % Nk == 4) {
            temp = sub_word(temp);
        }
        w[i] = w[i - Nk] ^ temp;
    }
}

// 轮密钥加
void add_round_key(uint8_t state[4][4], uint32_t *round_key) {
    for (int i = 0; i < 4; ++i) {
        uint32_t word = round_key[i];
        state[i][0] ^= (word >> 24) & 0xFF;
        state[i][1] ^= (word >> 16) & 0xFF;
        state[i][2] ^= (word >> 8) & 0xFF;
        state[i][3] ^= word & 0xFF;
    }
}

// 字节替换
void sub_bytes(uint8_t state[4][4]) {
    for (int i = 0; i < 4; ++i) {
        for (int j = 0; j < 4; ++j) {
            state[i][j] = sbox[state[i][j]];
        }
    }
}

// 行移位
void shift_rows(uint8_t state[4][4]) {
    // 第1行左移1位
    uint8_t temp = state[1][0];
    state[1][0] = state[1][1];
    state[1][1] = state[1][2];
    state[1][2] = state[1][3];
    state[1][3] = temp;

    // 第2行左移2位
    temp = state[2][0];
    state[2][0] = state[2][2];
    state[2][2] = temp;
    temp = state[2][1];
    state[2][1] = state[2][3];
    state[2][3] = temp;

    // 第3行左移3位
    temp = state[3][3];
    state[3][3] = state[3][2];
    state[3][2] = state[3][1];
    state[3][1] = state[3][0];
    state[3][0] = temp;
}

// 有限域GF(2^8)中的乘法
uint8_t gmul(uint8_t a, uint8_t b) {
    uint8_t p = 0;
    uint8_t hi_bit_set;
    for (int i = 0; i < 8; ++i) {
        if (b & 1) {
            p ^= a;
        }
        hi_bit_set = a & 0x80;
        a <<= 1;
        if (hi_bit_set) {
            a ^= 0x1b; // x^8 + x^4 + x^3 + x + 1
        }
        b >>= 1;
    }
    return p;
}

// 列混合
void mix_columns(uint8_t state[4][4]) {
    for (int i = 0; i < 4; ++i) {
        uint8_t a0 = state[i][0], a1 = state[i][1], a2 = state[i][2], a3 = state[i][3];
        state[i][0] = gmul(a0, 2) ^ gmul(a1, 3) ^ a2 ^ a3;
        state[i][1] = a0 ^ gmul(a1, 2) ^ gmul(a2, 3) ^ a3;
        state[i][2] = a0 ^ a1 ^ gmul(a2, 2) ^ gmul(a3, 3);
        state[i][3] = gmul(a0, 3) ^ a1 ^ a2 ^ gmul(a3, 2);
    }
}

// AES单块加密
void aes_encrypt_block(const uint8_t *input, uint8_t *output, uint32_t *w, int Nr) {
    uint8_t state[4][4];
    
    // 将输入复制到状态矩阵
    for (int i = 0; i < 4; ++i) {
        for (int j = 0; j < 4; ++j) {
            state[i][j] = input[j*4 + i];
        }
    }
    
    // 初始轮密钥加
    add_round_key(state, w);
    
    // 主加密循环
    for (int round = 1; round < Nr; ++round) {
        sub_bytes(state);
        shift_rows(state);
        mix_columns(state);
        add_round_key(state, w + round*4);
    }
    
    // 最后一轮（没有列混合）
    sub_bytes(state);
    shift_rows(state);
    add_round_key(state, w + Nr*4);
    
    // 将状态矩阵复制到输出
    for (int i = 0; i < 4; ++i) {
        for (int j = 0; j < 4; ++j) {
            output[j*4 + i] = state[i][j];
        }
    }
}

// 去除字符串中的所有空白字符
void remove_whitespace(std::string &str) {
    str.erase(std::remove_if(str.begin(), str.end(), ::isspace), str.end());
}

// 十六进制字符转数值
uint8_t hex_char_to_val(char c) {
    if (c >= '0' && c <= '9') return c - '0';
    if (c >= 'a' && c <= 'f') return c - 'a' + 10;
    if (c >= 'A' && c <= 'F') return c - 'A' + 10;
    return 0; // 无效字符，实际应用中应处理错误
}

// 十六进制字符串转字节数组
bool hex_to_bytes(const std::string &hex, uint8_t *bytes, size_t &len) {
    if (hex.length() % 2 != 0) return false;
    
    len = hex.length() / 2;
    if (len > 0 && bytes == nullptr) return false;
    
    for (size_t i = 0; i < len; ++i) {
        bytes[i] = (hex_char_to_val(hex[i*2]) << 4) | hex_char_to_val(hex[i*2+1]);
    }
    return true;
}

// 字节数组转十六进制字符串
std::string bytes_to_hex(const uint8_t *bytes, size_t len) {
    std::stringstream ss;
    ss << std::hex << std::setfill('0');
    for (size_t i = 0; i < len; ++i) {
        ss << std::setw(2) << static_cast<int>(bytes[i]);
    }
    return ss.str();
}

// AES-OFB加密函数
bool aes_ofb_encrypt(const uint8_t *plaintext, size_t plaintext_len, 
                    const uint8_t *key, size_t key_len, 
                    const uint8_t *iv, size_t iv_len, 
                    uint8_t *ciphertext, size_t &ciphertext_len) {
    // 检查输入参数
    if (plaintext == nullptr || key == nullptr || iv == nullptr || ciphertext == nullptr) {
        return false;
    }
    
    if (iv_len != AES_BLOCK_SIZE) {
        return false;
    }
    
    int Nr;
    if (key_len == 16) {
        Nr = AES_128_ROUNDS;
    } else if (key_len == 24) {
        Nr = AES_192_ROUNDS;
    } else if (key_len == 32) {
        Nr = AES_256_ROUNDS;
    } else {
        return false;
    }
    
    // 密钥扩展
    int w_len = (Nr + 1) * 4;
    uint32_t *w = new uint32_t[w_len];
    key_expansion(key, key_len, w);
    
    // 初始化反馈寄存器
    uint8_t feedback[AES_BLOCK_SIZE];
    memcpy(feedback, iv, AES_BLOCK_SIZE);
    
    // 加密
    ciphertext_len = plaintext_len;
    uint8_t keystream[AES_BLOCK_SIZE];
    
    for (size_t i = 0; i < plaintext_len; i += AES_BLOCK_SIZE) {
        // 生成密钥流
        aes_encrypt_block(feedback, keystream, w, Nr);
        
        // 更新反馈寄存器
        memcpy(feedback, keystream, AES_BLOCK_SIZE);
        
        // 计算密文块大小
        size_t block_size = (i + AES_BLOCK_SIZE > plaintext_len) 
                           (plaintext_len - i) : AES_BLOCK_SIZE;
        
        // 生成密文
        for (size_t j = 0; j < block_size; ++j) {
            ciphertext[i + j] = plaintext[i + j] ^ keystream[j];
        }
    }
    
    delete[] w;
    return true;
}

        remove_whitespace(key_hex);
    } else {
        std::cout << "请输入密钥(十六进制): ";

        remove_whitespace(key_hex);
    }

    // 处理IV
    if (env_iv && strlen(env_iv) > 0) {
        iv_hex = env_iv;
        remove_whitespace(iv_hex);
    } else {
        std::cout << "请输入IV(十六进制): ";
        std::getline(std::cin, iv_hex);
        remove_whitespace(iv_hex);
    }

    // 验证输入
    if (plaintext_hex.empty() || key_hex.empty() || iv_hex.empty()) {
        std::cerr << "输入不能为空" << std::endl;
        return 1;
    }

    // 转换为字节数组
    size_t plaintext_len, key_len, iv_len;
    uint8_t *plaintext = new uint8_t[plaintext_hex.length() / 2];
    uint8_t *key = new uint8_t[key_hex.length() / 2];
    uint8_t *iv = new uint8_t[iv_hex.length() / 2];

    if (!hex_to_bytes(plaintext_hex, plaintext, plaintext_len) ||
        !hex_to_bytes(key_hex, key, key_len) ||

        delete[] plaintext;
        delete[] key;
        delete[] iv;
        return 1;
    }

    // 验证密钥长度
    if (key_len != 16 && key_len != 24 && key_len != 32) {
        std::cerr << "密钥长度必须为16字节(128位)、24字节(192位)或32字节(256位)" << std::endl;
        delete[] plaintext;
        delete[] key;
        delete[] iv;
        return 1;
    }

    // 验证IV长度 (AES块大小为16字节)
    if (iv_len != 16) {
        std::cerr << "IV长度必须为16字节" << std::endl;
        delete[] plaintext;
        delete[] key;
        delete[] iv;
        return 1;
    }

    // 分配密文缓冲区
    uint8_t *ciphertext = new uint8_t[plaintext_len];
    size_t ciphertext_len;

    // 执行加密
    if (!aes_ofb_encrypt(plaintext, plaintext_len, key, key_len, iv, iv_len, ciphertext, ciphertext_len)) {
        std::cerr << "加密失败" << std::endl;
        delete[] plaintext;
        delete[] key;
        delete[] iv;
        delete[] ciphertext;
        return 1;
    }

    // 输出密文
    std::string ciphertext_hex = bytes_to_hex(ciphertext, ciphertext_len);
    std::cout << "密文: " << ciphertext_hex << std::endl;

    // 清理资源
    delete[] plaintext;
    delete[] key;
    delete[] iv;
    delete[] ciphertext;

    return 0;
}
